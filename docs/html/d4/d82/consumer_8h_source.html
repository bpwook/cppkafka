513 template <typename Allocator>
514 std::vector<Message, Allocator> Consumer::poll_batch(size_t max_batch_size,
515  std::chrono::milliseconds timeout,
516  const Allocator& alloc) {
517  std::vector<rd_kafka_message_t*> raw_messages(max_batch_size);
518  // Note that this will leak the queue when using rdkafka < 0.11.5 (see get_queue comment)
519  Queue queue = Queue::make_queue(rd_kafka_queue_get_consumer(get_handle()));
520  ssize_t result = rd_kafka_consume_batch_queue(queue.get_handle(),
521  timeout.count(),
522  raw_messages.data(),
523  raw_messages.size());
524  if (result == -1) {
525  check_error(rd_kafka_last_error());
526  // on the off-chance that check_error() does not throw an error
527  return std::vector<Message, Allocator>(alloc);
528  }
529  return std::vector<Message, Allocator>(raw_messages.begin(),
530  raw_messages.begin() + result,
531  alloc);
532 }
