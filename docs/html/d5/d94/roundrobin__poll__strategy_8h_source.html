149 template <typename Allocator>
150 std::vector<Message, Allocator> RoundRobinPollStrategy::poll_batch(size_t max_batch_size,
151  std::chrono::milliseconds timeout,
152  const Allocator& alloc) {
153  std::vector<Message, Allocator> messages(alloc);
154  ssize_t count = max_batch_size;
155 
156  // batch from the group event queue first (non-blocking)
157  consume_batch(get_consumer_queue().queue, messages, count, std::chrono::milliseconds(0), alloc);
158  size_t num_queues = get_partition_queues().size();
159  while ((count > 0) && (num_queues--)) {
160  // batch from the next partition (non-blocking)
161  consume_batch(get_next_queue().queue, messages, count, std::chrono::milliseconds(0), alloc);
162  }
163  // we still have space left in the buffer
164  if (count > 0) {
165  // wait on the event queue until timeout
166  consume_batch(get_consumer_queue().queue, messages, count, timeout, alloc);
167  }
168  return messages;
169 }

171 template <typename Allocator>
172 void RoundRobinPollStrategy::consume_batch(Queue& queue,
173  std::vector<Message, Allocator>& messages,
174  ssize_t& count,
175  std::chrono::milliseconds timeout,
176  const Allocator& alloc) {
177  std::vector<Message, Allocator> queue_messages = queue.consume_batch(count, timeout, alloc);
178  if (queue_messages.empty()) {
179  return;
180  }
181  // concatenate both lists
182  messages.insert(messages.end(),
183  make_move_iterator(queue_messages.begin()),
184  make_move_iterator(queue_messages.end()));
185  // reduce total batch count
186  count -= queue_messages.size();
187 }
