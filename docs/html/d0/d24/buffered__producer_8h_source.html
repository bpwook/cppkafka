604 template <typename BufferType, typename Allocator>
605 void BufferedProducer<BufferType, Allocator>::sync_produce(const MessageBuilder& builder) {
606  if (has_internal_data_) {
607  MessageBuilder builder_clone(builder.clone());
608  TrackerPtr tracker = add_tracker(SenderType::Sync, builder_clone);
609  // produce until we succeed or we reach max retry limit
610  std::future<bool> should_retry;
611  do {
612  should_retry = tracker->get_new_future();
613  produce_message(builder_clone);
614  wait_for_acks();
615  }
616  while (should_retry.get());
617  }
618  else {
619  // produce once
620  produce_message(builder);
621  wait_for_acks();
622  }
623 }

630 template <typename BufferType, typename Allocator>
631 void BufferedProducer<BufferType, Allocator>::async_flush() {
632  CounterGuard<size_t> counter_guard(flushes_in_progress_);
633  auto queue_flusher = [this](QueueType& queue, std::mutex & mutex)->void
634  {
635  QueueType flush_queue; // flush from temporary queue
636  swap_queues(queue, flush_queue, mutex);
637 
638  while (!flush_queue.empty()) {
639  async_produce(std::move(flush_queue.front()), false);
640  flush_queue.pop_front();
641  }
642  };
643  queue_flusher(retry_messages_, retry_mutex_);
644  queue_flusher(messages_, mutex_);
645  wait_for_acks(std::chrono::milliseconds(0)); //flush the producer but don't wait
646 }

648 template <typename BufferType, typename Allocator>
649 void BufferedProducer<BufferType, Allocator>::flush(bool preserve_order) {
650  if (preserve_order) {
651  CounterGuard<size_t> counter_guard(flushes_in_progress_);
652  auto queue_flusher = [this](QueueType& queue, std::mutex & mutex)->void
653  {
654  QueueType flush_queue; // flush from temporary queue
655  swap_queues(queue, flush_queue, mutex);
656 
657  while (!flush_queue.empty()) {
658  sync_produce(flush_queue.front());
659  flush_queue.pop_front();
660  }
661  };
662  queue_flusher(retry_messages_, retry_mutex_);
663  queue_flusher(messages_, mutex_);
664  }
665  else {
666  async_flush();
667  wait_for_acks();
668  }
669 }

671 template <typename BufferType, typename Allocator>
672 bool BufferedProducer<BufferType, Allocator>::flush(std::chrono::milliseconds timeout,
673  bool preserve_order) {
674  if (preserve_order) {
675  CounterGuard<size_t> counter_guard(flushes_in_progress_);
676  QueueType flush_queue; // flush from temporary queue
677  swap_queues(messages_, flush_queue, mutex_);
678  QueueType retry_flush_queue; // flush from temporary retry queue
679  swap_queues(retry_messages_, retry_flush_queue, retry_mutex_);
680 
681  auto queue_flusher = [this](QueueType& queue)->bool
682  {
683  if (!queue.empty()) {
684  sync_produce(queue.front());
685  queue.pop_front();
686  return true;
687  }
688  return false;
689  };
690  auto remaining = timeout;
691  auto start_time = std::chrono::high_resolution_clock::now();
692  do {
693  if (!queue_flusher(retry_flush_queue) && !queue_flusher(flush_queue)) {
694  break;
695  }
696  // calculate remaining time
697  remaining = timeout - std::chrono::duration_cast<std::chrono::milliseconds>
698  (std::chrono::high_resolution_clock::now() - start_time);
699  } while (remaining.count() > 0);
700 
701  // Re-enqueue remaining messages in original order
702  auto re_enqueuer = [this](QueueType& src_queue, QueueType& dst_queue, std::mutex & mutex)->void
703  {
704  if (!src_queue.empty()) {
705  std::lock_guard<std::mutex> lock(mutex);
706  dst_queue.insert(dst_queue.begin(),
707  std::make_move_iterator(src_queue.begin()),
708  std::make_move_iterator(src_queue.end()));
709  }
710  };
711  re_enqueuer(retry_flush_queue, retry_messages_, retry_mutex_);
712  re_enqueuer(flush_queue, messages_, mutex_);
713  return true;
714  }
715  else {
716  async_flush();
717  return wait_for_acks(timeout);
718  }
719 }

721 template <typename BufferType, typename Allocator>
722 void BufferedProducer<BufferType, Allocator>::wait_for_acks() {
723  while (pending_acks_ > 0) {
724  try {
725  producer_.flush();
726  }
727  catch (const HandleException& ex) {
728  // If we just hit the timeout, keep going, otherwise re-throw
729  if (ex.get_error() == RD_KAFKA_RESP_ERR__TIMED_OUT) {
730  continue;
731  }
732  else {
733  throw;
734  }
735  }
736  }
737 }

739 template <typename BufferType, typename Allocator>
740 bool BufferedProducer<BufferType, Allocator>::wait_for_acks(std::chrono::milliseconds timeout) {
741  auto remaining = timeout;
742  auto start_time = std::chrono::high_resolution_clock::now();
743  do {
744  try {
745  producer_.flush(remaining);
746  }
747  catch (const HandleException& ex) {
748  // If we just hit the timeout, keep going, otherwise re-throw
749  if (ex.get_error() == RD_KAFKA_RESP_ERR__TIMED_OUT) {
750  // There is no time remaining
751  return (pending_acks_ == 0);
752  }
753  else {
754  throw;
755  }
756  }
757  // calculate remaining time
758  remaining = timeout - std::chrono::duration_cast<std::chrono::milliseconds>
759  (std::chrono::high_resolution_clock::now() - start_time);
760  } while ((pending_acks_ > 0) && (remaining.count() > 0));
761  return (pending_acks_ == 0);
762 }

801 template <typename BufferType, typename Allocator>
802 template <typename BuilderType>
803 void BufferedProducer<BufferType, Allocator>::do_add_message(BuilderType&& builder,
804  QueueKind queue_kind,
805  FlushAction flush_action) {
806  if (queue_kind == QueueKind::Retry) {
807  std::lock_guard<std::mutex> lock(retry_mutex_);
808  retry_messages_.emplace_back(std::forward<BuilderType>(builder));
809  }
810  else {
811  std::lock_guard<std::mutex> lock(mutex_);
812  messages_.emplace_back(std::forward<BuilderType>(builder));
813  }
814 
815  // Flush the queues only if a regular message is added. Retry messages may be added
816  // from rdkafka callbacks, and flush/async_flush is a user-level call
817  if (queue_kind == QueueKind::Regular && flush_action == FlushAction::DoFlush && (max_buffer_size_ >= 0) && (max_buffer_size_ <= (ssize_t)get_buffer_size())) {
818  if (flush_method_ == FlushMethod::Sync) {
819  flush();
820  }
821  else {
822  async_flush();
823  }
824  }
825 }

917 template <typename BufferType, typename Allocator>
918 template <typename BuilderType>
919 void BufferedProducer<BufferType, Allocator>::produce_message(BuilderType&& builder) {
920  using builder_type = typename std::decay<BuilderType>::type;
921  bool queue_full_notify = queue_full_notification_ != QueueFullNotification::None;
922  while (true) {
923  try {
924  MessageInternalGuard<builder_type> internal_guard(const_cast<builder_type&>(builder));
925  producer_.produce(builder);
926  internal_guard.release();
927  // Sent successfully
928  ++pending_acks_;
929  break;
930  }
931  catch (const HandleException& ex) {
932  if (ex.get_error() == RD_KAFKA_RESP_ERR__QUEUE_FULL) {
933  // If the output queue is full, then just poll
934  producer_.poll();
935  // Notify application so it can slow-down production
936  if (queue_full_notify) {
937  queue_full_notify = queue_full_notification_ == QueueFullNotification::EachOccurence;
938  CallbackInvoker<QueueFullCallback>("queue full", queue_full_callback_, &producer_)(builder);
939  }
940  }
941  else {
942  throw;
943  }
944  }
945  }
946 }

948 template <typename BufferType, typename Allocator>
949 template <typename BuilderType>
950 void BufferedProducer<BufferType, Allocator>::async_produce(BuilderType&& builder, bool throw_on_error) {
951  try {
952  TestParameters* test_params = get_test_parameters();
953  if (test_params && test_params->force_produce_error_) {
954  throw HandleException(Error(RD_KAFKA_RESP_ERR_UNKNOWN));
955  }
956  produce_message(builder);
957  }
958  catch (const HandleException& ex) {
959  // If we have a flush failure callback and it returns true, we retry producing this message later
960  CallbackInvoker<FlushFailureCallback> callback("flush failure", flush_failure_callback_, &producer_);
961  if (!callback || callback(builder, ex.get_error())) {
962  TrackerPtr tracker = std::static_pointer_cast<Tracker>(builder.internal());
963  if (tracker && tracker->num_retries_ > 0) {
964  --tracker->num_retries_;
965  do_add_message(std::forward<BuilderType>(builder), QueueKind::Retry, FlushAction::DontFlush);
966  return;
967  }
968  }

987 template <typename BufferType, typename Allocator>
988 void BufferedProducer<BufferType, Allocator>::on_delivery_report(const Message& message) {
989  //Get tracker data
990  TestParameters* test_params = get_test_parameters();
991  TrackerPtr tracker = has_internal_data_ ?
992  std::static_pointer_cast<Tracker>(MessageInternal::load(const_cast<Message&>(message))->get_internal()) : nullptr;
993  bool should_retry = false;
994  if (message.get_error() || (test_params && test_params->force_delivery_error_)) {
995  // We should produce this message again if we don't have a produce failure callback
996  // or we have one but it returns true
997  CallbackInvoker<ProduceFailureCallback> callback("produce failure", produce_failure_callback_, &producer_);
998  if (!callback || callback(message)) {
999  // Check if we have reached the maximum retry limit
1000  if (tracker && tracker->num_retries_ > 0) {
1001  --tracker->num_retries_;
1002  if (tracker->sender_ == SenderType::Async) {
1003  // Re-enqueue for later retransmission with higher priority (i.e. front of the queue)
1004  do_add_message(Builder(message), QueueKind::Retry, FlushAction::DontFlush);
1005  }
1006  should_retry = true;
1007  }
1008  else {
1009  ++total_messages_dropped_;
1010  CallbackInvoker<ProduceTerminationCallback>
1011  ("produce termination", produce_termination_callback_, &producer_)(message);
1012  }
1013  }
1014  else {
1015  ++total_messages_dropped_;
1016  CallbackInvoker<ProduceTerminationCallback>
1017  ("produce termination", produce_termination_callback_, &producer_)(message);
1018  }
1019  }
1020  else {
1021  // Successful delivery
1022  CallbackInvoker<ProduceSuccessCallback>("delivery success", produce_success_callback_, &producer_)(message);
1023  // Increment the total successful transmissions
1024  ++total_messages_produced_;
1025  }
1026  // Signal producers
1027  if (tracker) {
1028  tracker->should_retry_.set_value(should_retry);
1029  }
1030  // Decrement the expected acks and check to prevent underflow
1031  if (pending_acks_ > 0) {
1032  --pending_acks_;
1033  }
1034 }
