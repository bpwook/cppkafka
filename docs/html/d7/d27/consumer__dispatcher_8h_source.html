132  template <typename Functor>
133  void handle_throttle(Throttle, const Functor& callback, Message msg) {
134  BackoffPerformer{}.perform([&]() {
135  if (!running_) {
136  return true;
137  }
138  msg = callback(std::move(msg));
139  if (msg) {
140  // Poll so we send heartbeats to the brokers
141  consumer_.poll();
142  }
143  return !msg;
144  });
145  }

287  template <typename Functor, typename... Functors>
288  auto process_message(const Functor& callback, Message msg, const Functors&... functors)
289  -> typename std::enable_if<std::is_same<Message, decltype(callback(std::move(msg)))>::value,
290  void>::type {
291  const auto throttle_ptr = &BasicConsumerDispatcher::handle_throttle<Functor>;
292  const auto default_throttler = std::bind(throttle_ptr, this, std::placeholders::_1,
293  std::placeholders::_2, std::placeholders::_3);
294 
295  using OnThrottleArgs = std::tuple<Throttle, const Functor&, Message>;
296  const auto on_throttle = find_matching_functor<OnThrottleArgs>(functors...,
297  default_throttler);
298 
299  msg = callback(std::move(msg));
300  // The callback rejected the message, start throttling
301  if (msg) {
302  // Pause consumption. When the pauser goes off scope, it will resume it
303  Pauser<ConsumerType> pauser(consumer_, consumer_.get_assignment());
304 
305  // Handle throttling on this message
306  on_throttle(Throttle{}, callback, std::move(msg));
307  }
308  }

332 template <typename ConsumerType>
333 template <typename... Args>
334 void BasicConsumerDispatcher<ConsumerType>::run(const Args&... args) {
335  using self = BasicConsumerDispatcher<ConsumerType>;
336 
337  // Make sure all callbacks match one of the signatures. Otherwise users could provide
338  // bogus callbacks that would never be executed
339  check_callbacks_match(args...);
340 
341  // This one is required
342  const auto on_message = find_matching_functor<OnMessageArgs>(args...);
343 
344  // For the rest, append our own implementation at the end as a fallback
345  const auto on_error = find_matching_functor<OnErrorArgs>(args..., &self::handle_error);
346  const auto on_eof = find_matching_functor<OnEofArgs>(args..., &self::handle_eof);
347  const auto on_timeout = find_matching_functor<OnTimeoutArgs>(args..., &self::handle_timeout);
348  const auto on_event = find_matching_functor<OnEventArgs>(args..., &self::handle_event);
349 
350  running_ = true;
351  while (running_) {
352  Message msg = consumer_.poll();
353  if (!msg) {
354  on_timeout(Timeout{});
355  }
356  else if (msg.get_error()) {
357  if (msg.is_eof()) {
358  on_eof(EndOfFile{}, { msg.get_topic(), msg.get_partition(), msg.get_offset() });
359  }
360  else {
361  on_error(msg.get_error());
362  }
363  }
364  else {
365  process_message(on_message, std::move(msg), args...);
366  }
367  on_event(Event{});
368  }
369 }
