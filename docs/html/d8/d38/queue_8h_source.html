249 template <typename Allocator>
250 std::vector<Message, Allocator> Queue::consume_batch(size_t max_batch_size,
251  std::chrono::milliseconds timeout,
252  const Allocator& alloc) const {
253  std::vector<rd_kafka_message_t*> raw_messages(max_batch_size);
254  ssize_t result = rd_kafka_consume_batch_queue(handle_.get(),
255  static_cast<int>(timeout.count()),
256  raw_messages.data(),
257  raw_messages.size());
258  if (result == -1) {
259  rd_kafka_resp_err_t error = rd_kafka_last_error();
260  if (error != RD_KAFKA_RESP_ERR_NO_ERROR) {
261  throw QueueException(error);
262  }
263  return std::vector<Message, Allocator>(alloc);
264  }
265  // Build message list
266  return std::vector<Message, Allocator>(raw_messages.begin(), raw_messages.begin() + result, alloc);
267 }
