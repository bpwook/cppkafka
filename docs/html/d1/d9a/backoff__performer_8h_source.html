118  template <typename Functor>
119  void perform(const Functor& callback) {
120  TimeUnit backoff = initial_backoff_;
121  size_t retries = maximum_retries_;
122  while (retries--) {
123  auto start = std::chrono::steady_clock::now();
124  // If the callback returns true, we're done
125  if (callback()) {
126  return;
127  }
128  auto end = std::chrono::steady_clock::now();
129  auto time_elapsed = end - start;
130  // If we still have time left, then sleep
131  if (time_elapsed < backoff) {
132  std::this_thread::sleep_for(backoff - time_elapsed);
133  }
134  // Increase out backoff depending on the policy being used
135  backoff = increase_backoff(backoff);
136  }
137  }
138 private:
139  TimeUnit increase_backoff(TimeUnit backoff);

143  TimeUnit maximum_backoff_;
144  BackoffPolicy policy_;
145  size_t maximum_retries_;
146 };
