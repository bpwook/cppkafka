274 template <typename Key, typename Value>
275 void CompactedTopicProcessor<Key, Value>::process_event() {
276  Message message = consumer_.poll();
277  if (message) {
278  if (!message.get_error()) {
279  boost::optional<Key> key = key_decoder_(message.get_key());
280  if (key) {
281  if (message.get_payload()) {
282  boost::optional<Value> value = value_decoder_(*key, message.get_payload());
283  if (value) {
284  // If there's a payload and we managed to parse the value, generate a
285  // SET_ELEMENT event
286  event_handler_({ Event::SET_ELEMENT, message.get_topic(),
287  message.get_partition(), *key, std::move(*value) });
288  }
289  }
290  else {
291  // No payload, generate a DELETE_ELEMENT event
292  event_handler_({ Event::DELETE_ELEMENT, message.get_topic(),
293  message.get_partition(), *key });
294  }
295  }
296  // Store the offset for this topic/partition
297  TopicPartition topic_partition(message.get_topic(), message.get_partition());
298  partition_offsets_[topic_partition] = message.get_offset();
299  }
300  else {
301  if (message.is_eof()) {
302  event_handler_({ Event::REACHED_EOF, message.get_topic(),
303  message.get_partition() });
304  }
305  else if (error_handler_) {
306  error_handler_(std::move(message));
307  }
308  }
309  }
310 }

312 template <typename K, typename V>
313 void CompactedTopicProcessor<K, V>::on_assignment(TopicPartitionList& topic_partitions) {
314  if (original_assignment_callback_) {
315  original_assignment_callback_(topic_partitions);
316  }
317  std::set<TopicPartition> partitions_found;
318  // See if we already had an assignment for any of these topic/partitions. If we do,
319  // then restore the offset following the last one we saw
320  for (TopicPartition& topic_partition : topic_partitions) {
321  auto iter = partition_offsets_.find(topic_partition);
322  if (iter != partition_offsets_.end()) {
323  topic_partition.set_offset(iter->second);
324  }
325  // Populate this set
326  partitions_found.insert(topic_partition);
327  }
328  // Clear our cache: remove any entries for topic/partitions that aren't assigned to us now.
329  // Emit a CLEAR_ELEMENTS event for each topic/partition that is gone
330  auto iter = partition_offsets_.begin();
331  while (iter != partition_offsets_.end()) {
332  const TopicPartition& topic_partition = iter->first;
333  if (partitions_found.count(topic_partition) == 0) {
334  event_handler_({ Event::CLEAR_ELEMENTS, topic_partition.get_topic(),
335  topic_partition.get_partition() });
336  iter = partition_offsets_.erase(iter);
337  }
338  else {
339  ++iter;
340  }
341  }
342 }
