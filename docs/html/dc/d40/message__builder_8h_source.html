242 template <typename T, typename C>
243 BasicMessageBuilder<T, C>::BasicMessageBuilder(const Message& message)
244 : topic_(message.get_topic()),
245  key_(Buffer(message.get_key().get_data(), message.get_key().get_size())),
246 #if (RD_KAFKA_VERSION >= RD_KAFKA_HEADERS_SUPPORT_VERSION)
247  //Here we must copy explicitly the Message headers since they are non-owning and this class
248  //assumes full ownership. Otherwise we will be holding an invalid handle when Message goes
249  //out of scope and rdkafka frees its resource.
250  header_list_(message.get_header_list() ?
251  HeaderListType(rd_kafka_headers_copy(message.get_header_list().get_handle())) : HeaderListType()), //copy headers
252 #endif
253  payload_(Buffer(message.get_payload().get_data(), message.get_payload().get_size())),
254  timestamp_(message.get_timestamp() ? message.get_timestamp().get().get_timestamp() :
255  std::chrono::milliseconds(0)),
256  user_data_(message.get_user_data()),
257  internal_(message.internal()) {
258 
259 }
